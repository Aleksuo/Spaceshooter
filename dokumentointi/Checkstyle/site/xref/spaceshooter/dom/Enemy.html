6   package spaceshooter.dom;
7   
8   import javax.swing.ImageIcon;
9   
10  /**
11   *
12   * @author Aleksi
13   */
14  public class Enemy extends ImageObject {
15  
16      public Enemy(int x, int y, int vx, int vy, int w, int h) {
17          super(x, y, vx, vy, w, h);
18          ImageIcon icon = new ImageIcon("./Resources/Sprites/vihollinen.png");
19          this.setSprite(icon.getImage());
20          this.setIsAlive(true);
21      }
22  
23      @Override
24      public void update(Level taso) {
25          move(taso.getPlayer());
26      }
27  
28      @Override
29      public void onCollision(GameObject obj) {
30          if (obj instanceof Projectile) {
31              this.setIsAlive(false);
32          }
33      }
34  
35      public void move(Player p) {
36          this.setVelX((p.getPosX() - this.getPosX()) / 30);
37          this.setVelY((p.getPosY() - this.getPosY()) / 30);
38  
39          this.setPosX(this.getPosX() + this.getVelX());
40          this.setPosY(this.getPosY() + this.getVelY());
41      }
42  
43  }
