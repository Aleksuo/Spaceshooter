6   package spaceshooter.dom;
7   
8   import javax.swing.ImageIcon;
9   
10  /**
11   * Enemy object that follows the player until it is destroyed or it collides
12   * with the player.
13   *
14   */
15  public class Mine extends Enemy {
16  
17      public Mine(float x, float y, int vx, int vy, int w, int h) {
18          super(x, y, vx, vy, w, h);
19          ImageIcon icon = new ImageIcon("./Resources/Sprites/miina.png");
20          this.setSprite(icon.getImage());
21          this.setHitpoints(125);
22          this.setScore(100);
23      }
24  
25      @Override
26      public void update(Level taso) {
27          move(taso.getPlayer());
28      }
29  
30      public void move(Player p) {
31          this.setVelX((p.getPosX() - this.getPosX()) / 30);
32          this.setVelY((p.getPosY() - this.getPosY()) / 30);
33  
34          this.setPosX(this.getPosX() + this.getVelX());
35          this.setPosY(this.getPosY() + this.getVelY());
36      }
37  
38      @Override
39      public void onCollision(GameObject obj) {
40          if (obj instanceof Projectile) {
41              Projectile projectile = (Projectile) obj;
42              int newHp = this.getHitpoints() - projectile.getDamage();
43              this.setHitpoints(newHp);
44  
45          }
46          if (this.getHitpoints() <= 0) {
47              this.setIsAlive(false);
48          } else if (obj instanceof Player) {
49              this.setIsAlive(false);
50          }
51  
52      }
53  
54  }
