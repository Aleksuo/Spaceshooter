6   package spaceshooter.dom;
7   
8   import java.awt.Point;
9   import javax.swing.ImageIcon;
10  
11  /**
12   * Object that can be controlled with user input.
13   *
14   */
15  public class Player extends ImageObject {
16  
17      private int ships;
18      private int charges;
19      private Weapon weapon;
20  
21      /**
22       * Constructor for Player.
23       *
24       * @param x x-coordinate.
25       * @param y y-coordinate.
26       * @param w width.
27       * @param h height.
28       */
29      public Player(int x, int y, int w, int h) {
30          super(x, y, 0, 0, w, h);
31          ImageIcon icon = new ImageIcon("./Resources/Sprites/alus.png");
32          this.setSprite(icon.getImage());
33          this.ships = 3;
34          this.charges = 3;
35          this.weapon = new Weapon(this, 5, 25);
36      }
37  
38      @Override
39      public void update(Level taso) {
40          Point piste = taso.getMousePosition();
41          if (piste != null) {
42              this.setPosX(piste.x);
43              this.setPosY(piste.y);
44          }
45  
46      }
47  
48      @Override
49      public void onCollision(GameObject obj) {
50          if (obj instanceof Enemy) {
51              this.destroyShip();
52          } else if (obj instanceof Update) {
53              this.weapon.upgrade();
54          }
55  
56      }
57  
58      /**
59       * Decreases players ship count.
60       */
61      private void destroyShip() {
62  
63          this.ships--;
64          if (this.ships < 1) {
65              this.setIsAlive(false);
66          }
67  
68          this.setPosX(0);
69          this.setPosY(0);
70      }
71  
72      public int getShips() {
73          return ships;
74      }
75  
76      public void setShips(int ships) {
77          this.ships = ships;
78      }
79  
80      public Weapon getWeapon() {
81          return weapon;
82      }
83  
84      public void setWeapon(Weapon weapon) {
85          this.weapon = weapon;
86      }
87  
88      public int getCharges() {
89          return charges;
90      }
91  
92      public void setCharges(int charges) {
93          this.charges = charges;
94      }
95  
96  }
