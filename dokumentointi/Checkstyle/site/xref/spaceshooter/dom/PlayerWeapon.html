20      @Override
21      public void spawnProjectiles(Level level) {
22          float x = this.getParent().getPosX() + 16;
23          float y = this.getParent().getPosY() - 10;
24          level.addObject(new PlayerProjectile(x, y, 0, -10, 10, 10, this.getDamage()));
25          if (this.getUpgrades() > 0) {
26              level.addObject(new PlayerProjectile(x, y, -2, -10, 10, 10, this.getDamage()));
27          }
28          if (this.getUpgrades() > 1) {
29              level.addObject(new PlayerProjectile(x, y, 2, -10, 10, 10, this.getDamage()));
30          }
31      }
