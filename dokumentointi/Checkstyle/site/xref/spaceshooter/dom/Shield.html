1   package spaceshooter.dom;
2   
3   import java.net.URL;
4   import javax.swing.ImageIcon;
5   
6   /**
7    * Absorbs taken damage for a certain time.
8    *
9    */
10  public class Shield extends ImageObject {
11  
12      private int ticks;
13      private GameObject parent;
14  
15      /**
16       * Constructor for Shield.
17       *
18       * @param x x-coordinate.
19       * @param y y-coordinate.
20       * @param vx x-velocity.
21       * @param vy y-velocity.
22       * @param w Hitbox widht.
23       * @param h Hitbox height.
24       * @param ticks Ticks alive.
25       * @param parent Parent of this object.
26       */
27      public Shield(float x, float y, float vx, float vy, int w, int h, int ticks, GameObject parent) {
28          super(x, y, vx, vy, w, h);
29          this.ticks = ticks;
30          URL url = this.getClass().getResource("/Sprites/shield.png");
31          ImageIcon icon = new ImageIcon(url);
32          this.setSprite(icon.getImage());
33          this.parent = parent;
34          this.parent.setCollisionOn(false);
35      }
36  
37      @Override
38      public void update(Level level) {
39          move();
40          tick();
41      }
42  
43      @Override
44      public void onDeath(Level level) {
45          this.parent.setCollisionOn(true);
46      }
47  
48      @Override
49      public void move() {
50          this.setPosX(this.parent.getPosX() - 16);
51          this.setPosY(this.parent.getPosY() - 16);
52      }
53  
54      /**
55       * Reduces ticks by one every frame.
56       */
57      public void tick() {
58          this.ticks--;
59          if (this.ticks < 1) {
60              this.setIsAlive(false);
61          }
62      }
63  
64      public int getTicks() {
65          return ticks;
66      }
67  
68      public void setTicks(int ticks) {
69          this.ticks = ticks;
70      }
71  
72      public GameObject getParent() {
73          return parent;
74      }
75  
76      public void setParent(GameObject parent) {
77          this.parent = parent;
78      }
79  
80  }
