6   package spaceshooter.dom.enemys;
7   
8   import java.net.URL;
9   import javax.swing.ImageIcon;
10  import spaceshooter.dom.enemys.Enemy;
11  import spaceshooter.dom.EnemyWeapon;
12  import spaceshooter.dom.GameObject;
13  import spaceshooter.dom.Level;
14  import spaceshooter.dom.Weapon;
15  
16  /**
17   * Enemy that moves and shoots projectiles towards the player.
18   *
19   */
20  public class Saucer extends Enemy {
21  
22      private int tick = 0;
23  
24      private Weapon weapon;
25  
26      /**
27       * Constructor for Saucer object.
28       *
29       * @param x x-coordinate.
30       * @param y y-coordinate.
31       * @param vx x-velocity.
32       * @param vy y-velocity.
33       * @param w width.
34       * @param h height.
35       */
36      public Saucer(float x, float y, float vx, float vy, int w, int h) {
37          super(x, y, vx, vy, w, h);
38          this.weapon = new EnemyWeapon(this, 2, 25);
39          URL url = this.getClass().getResource("/Sprites/Saucer.png");
40          ImageIcon icon = new ImageIcon(url);
41          this.setSprite(icon.getImage());
42          this.setHitpoints(100);
43          this.setScore(150);
44  
45      }
46  
47      @Override
48      public void move() {
49          tick++;
50          this.setPosX(this.getPosX() + 20 * (float) Math.cos(tick * 0.05 * Math.PI));
51          this.setPosY(this.getPosY() + this.getVelY());
52      }
53  
54      @Override
55      public void update(Level level) {
56          move();
57          this.weapon.shoot(level);
58      }
59  
60      @Override
61      public Enemy clone() {
62          return new Saucer(this.getPosX(), this.getPosY(), this.getVelX(),
63                  this.getVelY(), this.getWidth(), this.getHeight());
64      }
65  
66  }
