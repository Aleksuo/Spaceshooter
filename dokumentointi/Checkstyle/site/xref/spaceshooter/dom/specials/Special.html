6   package spaceshooter.dom.specials;
7   
8   import spaceshooter.dom.GameObject;
9   import spaceshooter.dom.Level;
10  import spaceshooter.util.Timer;
11  
12  /**
13   * Baseclass for Specials.
14   *
15   */
16  public class Special {
17  
18      private GameObject parent;
19      private int charges;
20  
21      private Timer timer;
22      private double lastFired;
23      private double delay;
24  
25      /**
26       * Constructor for Special.
27       *
28       * @param parent Parent of this Special.
29       * @param delay Delay between uses.
30       */
31      public Special(GameObject parent, double delay) {
32          this.parent = parent;
33          this.charges = 3;
34          this.delay = delay;
35          this.timer = new Timer();
36          this.lastFired = this.timer.elapsedTimeInSeconds();
37      }
38  
39      /**
40       * Triggers special.
41       *
42       * @param level Current level.
43       */
44      public void trigger(Level level) {
45          if (this.getParent().isAlive() && this.charges > 0) {
46              double deltaTime = this.timer.elapsedTimeInSeconds() - lastFired;
47              if (deltaTime > this.delay) {
48                  this.lastFired = timer.elapsedTimeInSeconds();
49                  discharge();
50                  spawnSpecial(level);
51              }
52          }
53      }
54  
55      /**
56       * Spawns special to the level.
57       *
58       * @param level Current level.
59       */
60      public void spawnSpecial(Level level) {
61      }
62  
63      /**
64       * Decreases charges by one. Called when special is triggered.
65       */
66      public void discharge() {
67          if (this.charges > 0) {
68              this.charges--;
69          }
70      }
71  
72      /**
73       * Increases charges by one.
74       */
75      public void charge() {
76          if (this.charges < 3) {
77              this.charges++;
78          }
79      }
80  
81      public GameObject getParent() {
82          return parent;
83      }
84  
85      public void setParent(GameObject parent) {
86          this.parent = parent;
87      }
88  
89      public int getCharges() {
90          return charges;
91      }
92  
93      public void setCharges(int charges) {
94          this.charges = charges;
95      }
96  
97      public Timer getTimer() {
98          return timer;
99      }
100 
101     public void setTimer(Timer timer) {
102         this.timer = timer;
103     }
104 
105     public double getLastFired() {
106         return lastFired;
107     }
108 
109     public void setLastFired(double lastFired) {
110         this.lastFired = lastFired;
111     }
112 
113     public double getDelay() {
114         return delay;
115     }
116 
117     public void setDelay(double delay) {
118         this.delay = delay;
119     }
120     
121     
122 
123 }
