6   package spaceshooter.dom.weapons;
7   
8   import spaceshooter.dom.GameObject;
9   import spaceshooter.dom.Level;
10  import spaceshooter.dom.Player;
11  import spaceshooter.dom.projectiles.EnemyProjectile;
12  
13  /**
14   * Weapon used by Enemys. Spawns EnemyProjectiles.
15   *
16   */
17  public class EnemyWeapon extends Weapon {
18  
19      /**
20       * Constructor for EnemyWeapon.
21       *
22       * @param obj Parent object.
23       * @param firerate Weapons firerate.
24       * @param damage Weapons damage.
25       */
26      public EnemyWeapon(GameObject obj, int firerate, int damage) {
27          super(obj, firerate, damage);
28      }
29  
30      @Override
31      public void spawnProjectiles(Level level) {
32          Player p = level.getPlayer();
33          float velX = (p.getPosX() - this.getParent().getPosX()) / 30;
34          float velY = (p.getPosY() - this.getParent().getPosY()) / 30;
35          level.addObject(new EnemyProjectile(this.getParent().getPosX(), this.getParent().getPosY(),
36                  velX, velY, 10, 10, this.getDamage()));
37      }
38  }
