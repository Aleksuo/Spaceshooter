6   package spaceshooter.managers;
7   
8   import java.util.HashMap;
9   import java.util.Map;
10  import spaceshooter.commands.SpawnEnemyCommand;
11  import spaceshooter.dom.enemys.Enemy;
12  import spaceshooter.dom.Level;
13  import spaceshooter.dom.enemys.Mine;
14  import spaceshooter.dom.Spawner;
15  import spaceshooter.dom.enemys.Saucer;
16  
17  /**
18   * Class that manages the spawners for enemys.
19   *
20   */
21  public class SpawnManager {
22  
23      private Map<Integer, Spawner> spawnerMap;
24      private Level level;
25  
26      /**
27       * Constructor for SpawnManager.
28       *
29       * @param level Current level.
30       */
31      public SpawnManager(Level level) {
32          this.level = level;
33          this.spawnerMap = new HashMap<Integer, Spawner>();
34          createSpawners();
35  
36      }
37  
38      /**
39       * Creates spawners for spawnable enemys.
40       */
41      public void createSpawners() {
42          this.spawnerMap.put(1, new Spawner(new Mine(0, 0, 10, 10, 32, 32)));
43          this.spawnerMap.put(2, new Spawner(new Saucer(0, 0, 5, 5, 32, 32)));
44      }
45  
46      /**
47       * Consumes SpawnEnemyCommands and spawns enemys.
48       *
49       * @param spec Command to be consumed.
50       */
51      public void handleSpawnCommand(SpawnEnemyCommand spec) {
52          Enemy enemy = this.spawnerMap.get(spec.getEnemyId())
53                  .spawnEnemy(spec.getX(), spec.getY());
54          this.level.addObject(enemy);
55      }
56  }
